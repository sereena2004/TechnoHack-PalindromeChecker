     for (i = 0; i < len / 2; i++) {
        if (s[i] != s[len - i - 1]) {
            flag = 1;
            break;
            }
    }
     if (flag) {
        printf("%s is not a palindrome\n", s);

        printf("%s is a palindrome\n", s);

}
